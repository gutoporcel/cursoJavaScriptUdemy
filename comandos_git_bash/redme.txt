git init // quando quer começar um projeto  que vamos  versionar



git pull //  atualiza o projeto  do remoto para o local

git status  // verifca  o status //

git add . //adiciona

git commit -a -m " descriçao breve do commit" // commitar o projeto  

git push // subir o projeto///


//criaçao de branch   

    git checkout -b "minhaNovaBreanch" // comando para criar nova Breanch
    git commit -a -m " comentario da nova branch"
    git push //  subir nova branch
//

// alterando entre as branch criadas

 git checkout nomeBranch
// 
// fazendo um merge
    git checkout nomeBranch //esta na que  vai entra o merge
    git branch // para  ter certeza em qual  branch eu to
    git merge nomeBranch //merge que vai se enviado para o principal
    git push
//

 git branch -d nomeBranch //  loca
